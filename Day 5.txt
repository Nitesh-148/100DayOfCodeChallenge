#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
      int grade;
      cin>>grade;
     if(grade>=38)
      {
        int rmd=grade%5;
        if(rmd>=3)
        grade+=5-rmd;
      }
      cout<<grade<<endl; 
    }  
  return 0; 
}
